1. Print Numbers 0 to 4 (While Loop)

Print numbers from 0 to 4 using a simple while loop.

#include <iostream>
using namespace std;

int main() {
    int i = 0;
    while (i < 5) {
        cout << i << endl;
        i++;
    }
    return 0;
}

2. Factorial of a Number

Calculate factorial using a while loop.

#include <iostream>
using namespace std;

int main() {
    int n;
    long long factorial = 1;
    cin >> n;

    while (n > 0) {
        factorial *= n;
        n--;
    }

    cout << factorial;
    return 0;
}

3. Reverse a Number

Reverse the digits of a given number using do-while loop.

#include <iostream>
using namespace std;

int main() {
    int num, rem;
    cin >> num;
    do {
        rem = num % 10;
        cout << rem;
        num /= 10;
    } while (num > 0);
    return 0;
}

4. Odd Numbers 100-50

Print all odd numbers between 100 and 50 using a for loop.

#include <iostream>
using namespace std;

int main() {
    for (int i = 100; i >= 50; i--) {
        if (i % 2 != 0) cout << i << endl;
    }
    return 0;
}

5. Find the Biggest of Three Numbers

Determine the biggest number among three integers.

#include <iostream>
using namespace std;

int main() {
    int A, B, C, biggest;
    char winner;
    cin >> A >> B >> C;

    if (A >= B && A >= C) { biggest = A; winner = 'A'; }
    else if (B >= C) { biggest = B; winner = 'B'; }
    else { biggest = C; winner = 'C'; }

    cout << winner << " = " << biggest;
    return 0;
}

6. Age & GPA Check

Check eligibility based on age and GPA using nested if statements.

#include <iostream>
using namespace std;

    cin >> age >> gpa;
    if(age >= 18){
